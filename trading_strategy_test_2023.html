# Create a function to signal when to buy and sell
# Modify the buy_sell() function to place the markers on the EMA crossover points
def buy_sell(data):
    sigPriceBuy = []
    sigPriceSell = []
    flag = -1

    for i in range(len(data)):
        if data['10ema'][i] > data['20ema'][i]:
            if flag != 1:
                # Calculate the crossover point of the EMAs
                crossover_point = data['20ema'][i] - (data['20ema'][i] - data['10ema'][i]) / 2

                # Use the crossover point as the buy signal
                sigPriceBuy.append(crossover_point)
                sigPriceSell.append(np.nan)
                flag = 1
            else:
                sigPriceBuy.append(np.nan)
                sigPriceSell.append(np.nan)

        elif data['10ema'][i] < data['20ema'][i]:
            if flag != 0:
                # Calculate the crossover point of the EMAs
                crossover_point = data['20ema'][i] + (data['10ema'][i] - data['20ema'][i]) / 2

                # Use the crossover point as the sell signal
                sigPriceBuy.append(np.nan)
                sigPriceSell.append(crossover_point)
                flag = 0
            else:
                sigPriceBuy.append(np.nan)
                sigPriceSell.append(np.nan)
        else:
            sigPriceBuy.append(np.nan)
            sigPriceSell.append(np.nan)

    return (sigPriceBuy, sigPriceSell)
